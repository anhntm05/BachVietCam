        function togglePassword() {
            const pwdInput = document.getElementById('password');
            const pwdIcon = document.getElementById('pwd-icon');
            if (pwdInput.type === 'password') {
                pwdInput.type = 'text';
                pwdIcon.innerText = 'visibility_off';
            } else {
                pwdInput.type = 'password';
                pwdIcon.innerText = 'visibility';
            }
        }

        // Add subtle mouse move effect to the glass panel
        const card = document.querySelector('.glass-panel');
        document.addEventListener('mousemove', (e) => {
            const x = e.clientX / window.innerWidth;
            const y = e.clientY / window.innerHeight;
            
            card.style.transform = `translate(${(x - 0.5) * 10}px, ${(y - 0.5) * 10}px)`;
        });

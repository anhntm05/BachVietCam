<script>
        // OTP Auto-focus Logic
        const inputs = document.querySelectorAll('#otp-container input');
        
        inputs.forEach((input, index) => {
            input.addEventListener('input', (e) => {
                if (e.target.value.length === 1 && index < inputs.length - 1) {
                    inputs[index + 1].focus();
                }
            });

            input.addEventListener('keydown', (e) => {
                if (e.key === 'Backspace' && !e.target.value && index > 0) {
                    inputs[index - 1].focus();
                }
            });
        });

        // Countdown Timer Logic
        let timeLeft = 119; // 1:59
        const timerDisplay = document.getElementById('timer');
        const countdownText = document.getElementById('countdown-text');
        const resendBtn = document.getElementById('resend-btn');

        const updateTimer = () => {
            const minutes = Math.floor(timeLeft / 60);
            const seconds = timeLeft % 60;
            timerDisplay.textContent = `${minutes.toString().padStart(2, '0')}:${seconds.toString().padStart(2, '0')}`;
            
            if (timeLeft <= 0) {
                clearInterval(timerInterval);
                countdownText.classList.add('hidden');
                resendBtn.classList.remove('hidden');
            }
            timeLeft--;
        };

        const timerInterval = setInterval(updateTimer, 1000);

        // Interaction for Resend Button
        resendBtn.addEventListener('click', () => {
            alert('Mã xác thực mới đã được gửi!');
            timeLeft = 119;
            resendBtn.classList.add('hidden');
            countdownText.classList.remove('hidden');
            // Restart interval logic would go here in a real app
        });

        // Form Submission
        document.getElementById('otp-form').addEventListener('submit', (e) => {
            e.preventDefault();
            const code = Array.from(inputs).map(i => i.value).join('');
            if (code.length === 6) {
                // Success state animation
                const btn = e.target.querySelector('button');
                btn.innerHTML = '<span class="material-symbols-outlined animate-spin" data-icon="progress_activity">progress_activity</span> Đang Xác Thực...';
                setTimeout(() => {
                    btn.innerHTML = '<span class="material-symbols-outlined" data-icon="check_circle">check_circle</span> Thành Công';
                    btn.classList.add('bg-secondary');
                }, 1500);
            } else {
                alert('Vui lòng nhập đầy đủ mã 6 chữ số.');
            }
        });
    </script>
